Transaktions API - synchron

def t_start(tname, il, conparams, iso, sql, sel="", printPre=False, printAfter=True):
    prefix = t_prefix(tname, il)
    print(f"{prefix}{iso.name}")
    con = psycopg.connect(conparams)
    con.set_isolation_level(iso)
    if printPre: print(f"{prefix}{con.execute(sel).fetchall()}")    
    con.execute(sql)
    print(f"{prefix}{sql}")
    if printAfter: print(f"{prefix}{con.execute(sel).fetchall()}")
    return con

def t_do(tname, il, con, sql, sel="", printPre=False, printAfter=True):
    prefix = t_prefix(tname, il)
    if printPre: print(f"{prefix}{con.execute(sel).fetchall()}")    
    print(f"{prefix}{sql}")
    con.execute(sql)
    if printAfter: print(f"{prefix}{con.execute(sel).fetchall()}")

def t_finish(tname, il, con, sel="", rollback=False, printState=False):
    prefix = t_prefix(tname, il)
    if printState: print(f"{prefix}{con.execute(sel).fetchall()}")    
    if rollback: 
        con.rollback()
        print(f"{prefix}rollback and close")
    else:
        con.commit()
        print(f"{prefix}commit and close")
    con.close()
