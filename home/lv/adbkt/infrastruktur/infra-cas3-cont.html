Cassandra Container 3 anlegen

Wichtig: Container cas1 muss bereits laufen.

Stack anlegen - Name: cas3

networks:
  ncas:
    name: ncas

services:
  cas1:
    container_name: cas3
    image: cassandra:latest
    ports:
      - 9044:9042
    environment:
      - CASSANDRA_SEEDS=cas1
      - JVM_OPTS=-Xms1024M -Xmx1024M
      - HEAP_NEWSIZE=1024M
      - MAX_HEAP_SIZE=1024M
    networks:
      - ncas
